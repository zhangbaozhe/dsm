Introduction: Shared Memory Model

A simple multi-threaded program


int main(int argc, char *argv[]) {
  std::mutex m;
  int i = 0;
  auto f1 = [&i, &m]() {
    for (int j = 0; j < 1000; j++) 
      i += 1; // Shared
  };
  auto f2 = [&i, &m]() {
    for (int j = 0; j < 1000; j++) {
      i += 1; // Not synchronized
    }
  };
  std::thread t1(f1); std::thread t2(f2);
  t1.join(); t2.join();
}
